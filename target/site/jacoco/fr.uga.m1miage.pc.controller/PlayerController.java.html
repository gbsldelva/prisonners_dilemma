package fr.uga.m1miage.pc.controller;

import fr.uga.m1miage.pc.model.Player;
import fr.uga.m1miage.pc.service.PlayerService;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/player")
public class PlayerController {
    private final PlayerService service;

    public PlayerController(PlayerService service) {
        this.service = service;
    }

    @PostMapping("/add")
    public Player connectPlayer(@RequestBody Player player) {
        return service.connectNewPlayer(player);
    }

    @GetMapping("/all")
    public List<Player> getAllPlayers() {
        return service.getAllPlayers();
    }

    @GetMapping("/find/{id}")
    public Player getPlayerByUsername(@PathVariable("id") String id) {
        return service.getPlayerByUsername(id);
    }

    @PutMapping("/update")
    public Player updatePlayer(@RequestBody Player player) {
        return service.updatePlayer(player);
    }

    @DeleteMapping("/disconnect/{id}")
    public void disconnectPlayer(@PathVariable("id") String id) {
        service.disconnectPlayer(id);
    }

    @GetMapping("allAvailable")
    List<Player> getAllAvailablePlayers() {
        return service.getAllAvailablePlayers();
    }
}
